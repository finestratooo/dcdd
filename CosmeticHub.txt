local function make(class, props, parent)
    local obj = Instance.new(class)
    for k, v in pairs(props) do
        obj[k] = v
    end
    if parent then obj.Parent = parent end
    return obj
end

local function corner(radius, parent)
    return make("UICorner", {CornerRadius = UDim.new(0, radius)}, parent)
end

local function stroke(thickness, color, parent)
    return make("UIStroke", {Thickness = thickness, Color = color, ApplyStrokeMode = Enum.ApplyStrokeMode.Border}, parent)
end

local function tween(obj, props, t, style, dir)
    TweenService:Create(obj,
        TweenInfo.new(t or 0.2, style or Enum.EasingStyle.Quart, dir or Enum.EasingDirection.Out),
        props
    ):Play()
end

local function hoverEffect(btn, normalColor, hoverColor)
    btn.MouseEnter:Connect(function()
        tween(btn, {BackgroundColor3 = hoverColor}, 0.15)
    end)
    btn.MouseLeave:Connect(function()
        tween(btn, {BackgroundColor3 = normalColor}, 0.15)
    end)
end

do
    local dragging, dragStart, startPos
    titleBar.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging  = true
            dragStart = input.Position
            startPos  = mainFrame.Position
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            local delta = input.Position - dragStart
            mainFrame.Position = UDim2.new(
                startPos.X.Scale,
                startPos.X.Offset + delta.X,
                startPos.Y.Scale,
                startPos.Y.Offset + delta.Y
            )
        end
    end)
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)
end

        BackgroundTransparency = 1,
        Visible          = false,
        ClipsDescendants = true,
    }, mainFrame)

    pages[name]      = page
    tabButtons[name] = btn

    btn.MouseButton1Click:Connect(function()
        if activeTab == name then return end
        -- Disattiva vecchia tab
        if activeTab then
            pages[activeTab].Visible = false
            tween(tabButtons[activeTab], {BackgroundColor3 = C.CARD, TextColor3 = C.MUTED}, 0.15)
        end
        activeTab = name
        page.Visible = true
        tween(btn, {BackgroundColor3 = C.ACCENT, TextColor3 = C.WHITE}, 0.15)
    end)

    return page
end

-- Pulsante Equipaggia
local statusLabel
local function setStatus(msg, color)
    statusLabel.Text       = msg
    statusLabel.TextColor3 = color or C.MUTED
end

local function getIdFromText(text)
    text = text:match("^%s*(.-)%s*$") -- trim
    -- URL catalogo: estrai numero
    local id = text:match("/catalog/(%d+)") or text:match("^(%d+)$")
    return tonumber(id)
end

local function equipById(id)
    if not id then
        setStatus("✕  ID non valido.", C.DANGER)
        return
    end
    setStatus("⏳  Caricamento asset " .. id .. "…", C.WARNING)

    task.spawn(function()
        local ok, asset = pcall(InsertService.LoadAsset, InsertService, id)
        if not ok or not asset then
            setStatus("✕  Asset non trovato o privato.", C.DANGER)
            return
        end

        local acc = asset:FindFirstChildOfClass("Accessory")
        if not acc then
            asset:Destroy()
            setStatus("✕  L'asset non è un accessorio.", C.DANGER)
            return
        end

        -- Equipaggia
        acc.Parent = workspace
        local char = localPlayer.Character
        local hum  = char and char:FindFirstChildOfClass("Humanoid")
        if hum then
            hum:AddAccessory(acc)
            -- Salva in lista
            table.insert(equippedList, {id = id, name = acc.Name, ref = acc})
            setStatus("✓  Equipaggiato: " .. acc.Name, C.SUCCESS)
            -- Aggiorna lista equipped
            if _G.refreshEquipped then _G.refreshEquipped() end
        else
            acc:Destroy()
            setStatus("✕  Personaggio non trovato.", C.DANGER)
        end
        asset:Destroy()
    end)
end
